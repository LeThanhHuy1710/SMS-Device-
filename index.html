async function load(){ try { const res = await fetch(BASE + "devices?select=*&order=created_at.desc", { headers: { "apikey": KEY, "Authorization": "Bearer " + KEY } }); if (!res.ok) { console.log("API ERROR:", res.status); return; } const data = await res.json(); console.log("DEVICES:", data); devices = Array.isArray(data) ? data : []; render(); } catch(e){ console.log("LOAD ERROR:", e); } } function render(){ const box = document.getElementById("list"); if (!box) { console.log("Không tìm thấy #list"); return; } box.innerHTML = ""; const now = Date.now(); devices.forEach(d => { // 👉 fix last_seen null / string const last = Number(d.last_seen) || 0; const isOnline = (now - last) < 15000; // 👉 fix thiếu dữ liệu const name = d.device_name || d.device || "Unknown device"; const id = d.id || ""; const div = document.createElement("div"); div.className = "device " + (isOnline ? "online" : "offline"); // 👉 dùng template string chuẩn (KHÔNG lỗi dấu `) div.innerHTML = `
${name}
${isOnline ? "🟢 Online" : "🔴 Offline"}
`; box.appendChild(div); }); }
